const express = require("express"); const http = require("http"); const { Server } = require("socket.io"); const app = express(); const server = http.createServer(app); const io = new Server(server); app.use(express.static("public")); io.on("connection", (socket) => { console.log("A user connected"); // Message received socket.on("chat:msg", (text) => { const safeText = String(text ?? "").slice(0, 2000); if (!safeText.trim()) return; io.emit("chat:msg", { text: safeText, ts: Date.now() }); }); // Typing events socket.on("chat:typing", () => { socket.broadcast.emit("chat:typing"); }); socket.on("chat:stopTyping", () => { socket.broadcast.emit("chat:stopTyping"); }); socket.on("disconnect", () => { console.log("A user disconnected"); }); }); server.listen(3000, () => { console.log("Server running at http://localhost:3000"); });
